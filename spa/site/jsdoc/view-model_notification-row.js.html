/**
 * ViewModel que conté les dades necessaries per pintar la vista d'una fila de la bústia de missatges.
 * @module view-model/notification-row
 * @see {@link module:view/notification-row} és la vista relacionada
 */

define([], function () {

    var NotificationRowViewModel = Backbone.Model.extend({
        /*
         * -----------------------------------------------
         * Valors per defecte
         * -----------------------------------------------
         */
        defaults: {
            'ico_result_class': null,
            'ico_action': null,
            'ico_action_class': null,
            'class_action': null,
            'descripcio': null,
            'llegit': null,
        },


        /*
         * -----------------------------------------------
         * Inicialització
         * -----------------------------------------------
         */
        initialize: function (model, options) {
            this.model = options.model;
            this.listenTo(this.model, 'change', this.onModelChange);

            this.onModelChange();
        },
        /*
         * -----------------------------------------------
         * Overwrites
         * -----------------------------------------------
         */

        /*
         * -----------------------------------------------
         * Accions
         * -----------------------------------------------
         */


        /*
         * -----------------------------------------------
         * Gestió d'events
         * -----------------------------------------------
         */
        onModelChange: function () {

            var estat = this.model.get('estat'),
                tipus = this.model.get('tipus');

            this.set(_.extend({
                    'descripcio': this.model.get('descripcio'),
                    'llegit': this.model.get('llegit'),
                }, NotificationRowViewModel.NOTIFICATION_TYPE_DICTIONARY[tipus],
                NotificationRowViewModel.NOTIFICATION_STATUS_DICTIONARY[estat]));
        },

        /*
         * -----------------------------------------------
         * Mètodes privats
         * -----------------------------------------------
         */
        notificationReaded: function () {
        	if (this.get('estat') == "pending") {
        		this.model.beforeExecutionTaskModal();
        	}
        	
        	else {
        		this.model.afterBeginExecutionTaskModal();
        	}
        	
        },

    }, {
        NOTIFICATION_STATUS_DICTIONARY: {
            'working': {
                'ico_action': '/img/notificacions/notificacio_treballant.svg',
                'ico_action_class': 'working gear',
                'class_action': '',
            },
            'success': {
                'ico_action': '/img/notificacions/notificacio_ok.svg',
                'ico_action_class': '',
                'class_action': '',
            },
            'error': {
                'ico_action': '/img/notificacions/notificacio_ko.svg',
                'ico_action_class': '',
                'class_action': '',
            }
        },
        NOTIFICATION_TYPE_DICTIONARY: {
            'envia_carret': {
                'ico_result_class': 'icon-carreto_fletxa',
            }
        }
    });
    return NotificationRowViewModel;
});
